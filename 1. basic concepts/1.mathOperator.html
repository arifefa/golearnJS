BASIC CONCEPTS

MATH OPERATORS 

    +, -, *, /, %, ++(increment), --(decrement) 
    
    var mathOperator = 7 + 3;
    document.write(mathOperator);

ASSIGMENT OPERATORS 

    =    ==>  X=Y      ==>   Y=X
    +=   ==>  X += Y   ==>   X=X+Y 
    -=   ==>  X -= Y   ==>   X=X-Y
    *=   ==>  X *= Y   ==>   X=X*Y
    /=   ==>  X /= Y   ==>   X=X/Y
    %=   ==>  X %= Y   ==>   X=X%Y  
    
    var assigmentOperator = 2 ;
    assigmentOperator += 5;
    document.write(assigmentOperator);

COMPARISON OPERATORS 

    ==    ==>  5 == 10   ==> FALSE
    ===   ==>  5 === 10  ==> FALSE
    !=    ==>  5 != 10   ==> TRUE
    !==   ==>  5 !== 10  ==> TRUE (same type)
    <     ==>  5 < 10    ==> TRUE
    <=    ==>  5 <= 10   ==> TRUE
    >     ==>  5 > 10    ==> FALSE
    >=    ==>  5 >= 10   ==> FALSE
    
    document.write(5 !== 7)

LOGICAL OPERATOR 

    && ==> and  ==>  TRUE if both operands are true
    || ==> or   ==>  TRUE if one of operands is true
    !  ==> not  ==>  TRUE if the operands is false
    
    document.write((5 < 7) && (12 > 7))
